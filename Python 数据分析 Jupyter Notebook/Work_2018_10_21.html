import numpy as np
import matplotlib.pyplot as plt

population_data = np.load("../data/populations.npz") # 
name = population_data["feature_names"] # 
values = population_data["data"][0:20] # 
plt.rcParams["font.sans-serif"] = "SimHei" # 
plt.rcParams["axes.unicode_minus"] = False

# 

p1 = plt.figure(figsize=(12,5))
ax1 = p1.add_subplot(1,2,1)
plt.bar(np.arange(0,20)-0.25,values[::-1,2],width=0.5) # 
plt.bar(np.arange(0,20)+0.25,values[::-1,3],width=0.5)
plt.xlabel("") # 
plt.ylabel("") # y
plt.xticks(range(20),values[::-1,0],rotation=45)
plt.legend(["",""]) # 
plt.title("1996~2015")

# 

ax2 = p1.add_subplot(1,2,2)
plt.bar(np.arange(0,20)-0.25,values[::-1,4],width=0.5) # 
plt.bar(np.arange(0,20)+0.25,values[::-1,5],width=0.5)
plt.xlabel("") # 
plt.ylabel("") # y
plt.xticks(range(20),values[::-1,0],rotation=45)
plt.legend(["",""]) # 
plt.title("1996~2015")
plt.savefig("../tmp/1996~2015.png")

# 

p2 = plt.figure(figsize=(12,5))
ax3 = p2.add_subplot(1,2,1)
plt.pie(values[0,(2,3)],explode=[0.01,0.01],labels=["",""],autopct="%1.1f%%") # 
plt.title("1996~2015")

# 

ax4 = p2.add_subplot(1,2,2)
plt.pie(values[0,(4,5)],explode=[0.01,0.01],labels=["",""],autopct="%1.1f%%")
plt.title("1996~2015")
plt.savefig("../tmp/1996~2015.png")

# 

p3 = plt.figure(figsize=(12,5))
ax5 = p3.add_subplot(1,2,1)
gdp1 = (list(values[::-1,2]),list(values[::-1,3]))
label1 = ["",""]
plt.boxplot(gdp1,notch=True,labels=label1,meanline=True)
plt.ylabel("")
plt.title("1996~2015")

# 

ax6 = p3.add_subplot(1,2,2)
gdp2 = (list(values[::-1,4]),list(values[::-1,5]))
label2 = ["",""]
plt.boxplot(gdp2,notch=True,labels=label2,meanline=True)
plt.ylabel("")
plt.title("1996~2015")
plt.savefig("../tmp/1996~2015.png")

plt.show()
